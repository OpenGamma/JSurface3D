  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.SurfacePlane requires JSurface3D');
  7     /**
  8      * Creates a surface plane with extrutions and vertex shading.
  9      * Clips plane if required
 10      * Adds update method
 11      * @name JSurface3D.SurfacePlane
 12      * @namespace JSurface3D.SurfacePlane
 13      * @param {Object} js3d JSurface3D instance
 14      * @private
 15      * @returns {Function} {THREE.Object3D} A surface plane object
 16      * @constructor
 17      */
 18     window.JSurface3D.SurfacePlane = function (js3d) {
 19         var surfaceplane = new THREE.Object3D(), wiremesh, planemesh;
 20         surfaceplane.name = 'SurfacePlane';
 21         surfaceplane.init = function () {
 22             var settings = js3d.settings, matlib = js3d.matlib, plane = JSurface3D.plane(js3d, 'surface'), i, wire;
 23             plane.verticesNeedUpdate = true;
 24             for (i = 0; i < js3d.adjusted_vol.length; i++) {plane.vertices[i].y = js3d.adjusted_vol[i];} // extrude
 25             wire = plane.clone();
 26             plane.computeCentroids();
 27             plane.computeFaceNormals();
 28             plane.computeBoundingSphere();
 29             (function () { // apply heatmap
 30                 if (!Detector.webgl) return;
 31                 var faces = 'abcd', face, color, vertex, index, i, k,
 32                     min = Math.min.apply(null, js3d.adjusted_vol), max = Math.max.apply(null, js3d.adjusted_vol),
 33                     hue_min = settings.vertex_shading_hue_min, hue_max = settings.vertex_shading_hue_max, hue;
 34                 for (i = 0; i < plane.faces.length; i ++) {
 35                     face = plane.faces[i];
 36                     for (k = 0; k < 4; k++) {
 37                         index = face[faces.charAt(k)];
 38                         vertex = plane.vertices[index];
 39                         color = new THREE.Color(0xffffff);
 40                         hue = ~~((vertex.y - min) / (max - min) * (hue_max - hue_min) + hue_min) / 360;
 41                         color.setHSV(hue, 0.9, 1);
 42                         face.vertexColors[k] = color;
 43                     }
 44                 }
 45             }());
 46             /* apply surface materials */
 47             (function () { // clip/slice
 48                 var row, i, x, l,
 49                     zlft = Math.abs(js3d.slice.lft_z_handle_position - js3d.z_segments) * js3d.x_segments,
 50                     zrgt = Math.abs(js3d.slice.rgt_z_handle_position - js3d.z_segments) * js3d.x_segments,
 51                     xlft = Math.abs(js3d.slice.lft_x_handle_position - js3d.x_segments),
 52                     xrgt = Math.abs(js3d.slice.rgt_x_handle_position - js3d.x_segments),
 53                     zmin = Math.min.apply(null, [zlft, zrgt]),
 54                     zmax = Math.max.apply(null, [zlft, zrgt]),
 55                     xmin = Math.min.apply(null, [xlft, xrgt]),
 56                     xmax = Math.max.apply(null, [xlft, xrgt]);
 57                 for (i = 0, x = 0, row = 0, l = plane.faces.length; i < l; i++, x++) {
 58                     var plane_face = plane.faces[i], wire_face = wire.faces[i];
 59                     if (x === js3d.x_segments) x = 0, row++;
 60                     if (
 61                         (i < zmax) && (i > zmin - 1) && // z slice
 62                         (i > xmin + row * js3d.x_segments -1) && (i < xmax + row * js3d.x_segments)  // x slice
 63                     ) plane_face.materialIndex = wire_face.materialIndex = 1;
 64                     else plane_face.materialIndex = wire_face.materialIndex = 0;
 65                 }
 66             }());
 67             wiremesh = new THREE.Mesh(wire, new THREE.MeshFaceMaterial(matlib.get_material('compound_surface_wire')));
 68             wiremesh.position.y = 0.01; // move wiremesh to account for Z-fighting
 69             planemesh = new THREE.Mesh(plane, new THREE.MeshFaceMaterial(matlib.get_material('compound_surface')));
 70             surfaceplane.add(planemesh);
 71             surfaceplane.add(wiremesh);
 72             surfaceplane.position.y = settings.floating_height;
 73             surfaceplane.matrixAutoUpdate = false;
 74             surfaceplane.updateMatrix();
 75             surfaceplane.children.forEach(function (mesh) {
 76                 mesh.doubleSided = true;
 77                 mesh.matrixAutoUpdate = false;
 78             });
 79             js3d.interactive_meshes.add('surface', surfaceplane.children[0]);
 80             if (js3d.buffers.surface) js3d.buffers.surface.add(planemesh);
 81             if (js3d.buffers.surface) js3d.buffers.surface.add(wiremesh);
 82         };
 83         surfaceplane.update = function () {
 84             if (js3d.buffers.surface) js3d.buffers.surface.clear(planemesh);
 85             if (js3d.buffers.surface) js3d.buffers.surface.clear(wiremesh);
 86             surfaceplane.remove(wiremesh);
 87             surfaceplane.remove(planemesh);
 88             surfaceplane.init(js3d);
 89         };
 90         return surfaceplane
 91     };
 92 })();
