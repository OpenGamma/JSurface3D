  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.Plane requires JSurface3D');
  7     /**
  8      * Creates a plane with the correct segments and spacing
  9      * @name JSurface3D.plane
 10      * @namespace JSurface3D.plane
 11      * @param {Object} js3d A JSurface3D instance
 12      * @param {String} type The type of plane to create. 'surface', 'smilex' or 'smiley'
 13      * @function
 14      * @private
 15      * @returns {THREE.PlaneGeometry}
 16      */
 17     window.JSurface3D.plane = function (js3d, type) {
 18         var xlen, ylen, xseg, yseg, xoff, yoff, plane, vertex, len, i, k, settings = js3d.settings;
 19         if (type === 'surface') {
 20             xlen = settings.surface_x;
 21             ylen = settings.surface_z;
 22             xseg = js3d.x_segments;
 23             yseg = js3d.z_segments;
 24             xoff = js3d.adjusted_xs;
 25             yoff = js3d.adjusted_zs;
 26         }
 27         if (type === 'smilex') {
 28             xlen = settings.surface_x;
 29             ylen = settings.surface_y;
 30             xseg = js3d.x_segments;
 31             yseg = js3d.y_segments;
 32             xoff = js3d.adjusted_xs;
 33             yoff = js3d.adjusted_ys;
 34         }
 35         if (type === 'smiley') {
 36             xlen = settings.surface_y;
 37             ylen = settings.surface_z;
 38             xseg = js3d.y_segments;
 39             yseg = js3d.z_segments;
 40             xoff = js3d.adjusted_ys;
 41             yoff = js3d.adjusted_zs;
 42         }
 43         plane = new THREE.PlaneGeometry(xlen, ylen, xseg, yseg);
 44         plane.applyMatrix(new THREE.Matrix4().makeRotationX(-Math.PI / 2));
 45         len = (xseg + 1) * (yseg + 1);
 46         for (i = 0, k = 0; i < len; i++, k++) {
 47             vertex = plane.vertices[i];
 48             if (typeof xoff[k] === 'undefined') k = 0;
 49             vertex.x = xoff[k];
 50             vertex.z = yoff[Math.floor(i / xoff.length)];
 51         }
 52         return plane;
 53     };
 54 })();
