  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.Slice requires JSurface3D');
  7     /**
  8      * Slice handle constructor
  9      * @return {THREE.Mesh}
 10      */
 11     var Handle = function (matlib, settings) {
 12         var geo = new THREE.CubeGeometry(3, 1.2, 3, 2, 0, 0);
 13         geo.vertices // move middle vertices out to a point
 14             .filter(function (val) {return (val.x === 0 && val.z === -1.5);})
 15             .forEach(function (vertex) {vertex.z = -2.9;});
 16         return new THREE.Mesh(geo, matlib.get_material('phong', settings.slice_handle_color));
 17     };
 18     /**
 19      * Slice Bar constructor
 20      * @param {Object} matlib material library
 21      * @param {Object} settings settings
 22      * @param {String} orientation 'x' or 'z'
 23      * @return {THREE.Mesh}
 24      */
 25     var SliceBar = function (matlib, settings, orientation) {
 26         var geo = new THREE.CubeGeometry(settings['surface_' + orientation], 1, 2, 0, 0),
 27             mesh = new THREE.Mesh(geo, matlib.get_material('phong', settings.slice_bar_color));
 28         if (orientation === 'x') mesh.position.z = settings.surface_z / 2 + 1.5 + settings.axis_offset;
 29         if (orientation === 'z') {
 30             mesh.position.x = -(settings.surface_x / 2) - 1.5 - settings.axis_offset;
 31             mesh.rotation.y = -Math.PI * 0.5;
 32         }
 33         mesh.matrixAutoUpdate = false;
 34         mesh.updateMatrix();
 35         return mesh;
 36     };
 37     /**
 38      * Create slicing functionality
 39      * @param {Object} js3d An instance of JSurface3D
 40      * @name JSurface3D.Slice
 41      * @mamespace JSurface3D.Slice
 42      * @private
 43      * @constructor
 44      */
 45     window.JSurface3D.Slice = function (js3d) {
 46         var slice = this, matlib = js3d.matlib, settings = js3d.settings;
 47         slice.lft_x_handle_position = js3d.x_segments;
 48         slice.rgt_x_handle_position = 0;
 49         slice.lft_z_handle_position = js3d.z_segments;
 50         slice.rgt_z_handle_position = 0;
 51         slice.load = function () {
 52             var plane = new THREE.PlaneGeometry(5000, 5000, 0, 0), mesh;
 53             plane.applyMatrix(new THREE.Matrix4().makeRotationX(-Math.PI / 2));
 54             mesh = new THREE.Mesh(plane, matlib.get_material('wire', 0xcccccc));
 55             mesh.matrixAutoUpdate = false;
 56             mesh.updateMatrix();
 57             slice.intersection_plane = mesh;
 58             js3d.groups.surface_bottom.add(slice.intersection_plane);
 59             slice.x();
 60             slice.z();
 61         };
 62         slice.reset_handle_material = function () {
 63             slice.lft_x_handle.material = slice.rgt_x_handle.material = slice.lft_z_handle.material =
 64             slice.rgt_z_handle.material = matlib.get_material('phong', settings.slice_handle_color);
 65         };
 66         slice.create_slice_bar = function (axis) {
 67             var vertices, bar_lbl = axis + '_bar',
 68                 lx = slice['lft_' + axis + '_handle'].position[axis],
 69                 rx = slice['rgt_' + axis + '_handle'].position[axis];
 70             if (js3d.buffers.slice) js3d.buffers.slice.clear(slice[bar_lbl]);
 71             js3d.groups.slice.remove(slice[bar_lbl]);
 72             slice[bar_lbl] = new SliceBar(matlib, settings, axis);
 73             vertices = slice[bar_lbl].geometry.vertices;
 74             vertices[0].x = vertices[1].x = vertices[2].x = vertices[3].x = Math.max.apply(null, [lx, rx]);
 75             vertices[4].x = vertices[5].x = vertices[6].x = vertices[7].x = Math.min.apply(null, [lx, rx]);
 76             if (!(Math.abs(lx) + Math.abs(rx) === settings['surface_' + axis])) // change color
 77                 slice[bar_lbl].material = matlib.get_material('phong', settings.slice_bar_color_active);
 78             js3d.groups.slice.add(js3d.buffers.slice.add(slice[bar_lbl]));
 79         };
 80         slice.x = function () {
 81             var xpos = (settings.surface_x / 2), zpos = settings.surface_z / 2 + 1.5 + settings.axis_offset;
 82             /**
 83              * particle guide
 84              * (dotted lines that guide the slice handles)
 85              */
 86             (function () {
 87                 var geo = new THREE.Geometry(), num_vertices = js3d.adjusted_xs.length, xparticles;
 88                 while (num_vertices--) geo.vertices.push(new THREE.Vector3(js3d.adjusted_xs[num_vertices], 0, 0));
 89                 xparticles = new THREE.ParticleSystem(geo, matlib.get_material('particles'));
 90                 xparticles.position.x += 0.1;
 91                 xparticles.position.z = zpos;
 92                 js3d.groups.surface_bottom.add(slice.x_particles = xparticles);
 93             }());
 94             /**
 95              * handles
 96              */
 97             if (!slice.lft_x_handle) slice.lft_x_handle = new Handle(matlib, settings);
 98             slice.lft_x_handle.position.x = -xpos;
 99             slice.lft_x_handle.position.z = zpos;
100             if (!slice.rgt_x_handle) slice.rgt_x_handle = new Handle(matlib, settings);
101             slice.rgt_x_handle.position.x = xpos;
102             slice.rgt_x_handle.position.z = zpos;
103             js3d.groups.surface_bottom.add(slice.lft_x_handle);
104             js3d.groups.surface_bottom.add(slice.rgt_x_handle);
105             js3d.interactive_meshes.add('lft_x_handle', slice.lft_x_handle);
106             js3d.interactive_meshes.add('rgt_x_handle', slice.rgt_x_handle);
107             slice.lft_x_handle.position.x = slice.x_particles.geometry.vertices[slice.lft_x_handle_position].x;
108             slice.rgt_x_handle.position.x = slice.x_particles.geometry.vertices[slice.rgt_x_handle_position].x;
109             /**
110              * slice bar
111              */
112             slice.create_slice_bar('x');
113         };
114         slice.z = function () {
115             var xpos = (settings.surface_x / 2) + 1.5 + settings.axis_offset, zpos = settings.surface_z / 2;
116             /**
117              * particle guide
118              * (dotted lines that guide the slice handles)
119              */
120             (function () {
121                 var geo = new THREE.Geometry(), num_vertices = js3d.adjusted_zs.length, zparticles;
122                 while (num_vertices--) geo.vertices.push(new THREE.Vector3(js3d.adjusted_zs[num_vertices], 0, 0));
123                 zparticles = new THREE.ParticleSystem(geo, matlib.get_material('particles'));
124                 zparticles.rotation.y = -Math.PI * 0.5;
125                 zparticles.position.x = -xpos;
126                 js3d.groups.surface_bottom.add(slice.z_particles = zparticles);
127             }());
128             /**
129              * handles
130              */
131             if (!slice.lft_z_handle) slice.lft_z_handle = new Handle(matlib, settings);
132             slice.lft_z_handle.position.x = -xpos;
133             slice.lft_z_handle.position.z = -zpos;
134             slice.lft_z_handle.rotation.y = -Math.PI * 0.5;
135             if (!slice.rgt_z_handle) slice.rgt_z_handle = new Handle(matlib, settings);
136             slice.rgt_z_handle.position.x = -xpos;
137             slice.rgt_z_handle.position.z = zpos;
138             slice.rgt_z_handle.rotation.y = -Math.PI * 0.5;
139             js3d.groups.surface_bottom.add(slice.lft_z_handle);
140             js3d.groups.surface_bottom.add(slice.rgt_z_handle);
141             js3d.interactive_meshes.add('lft_z_handle', slice.lft_z_handle);
142             js3d.interactive_meshes.add('rgt_z_handle', slice.rgt_z_handle);
143             slice.lft_z_handle.position.z = slice.z_particles.geometry.vertices[slice.lft_z_handle_position].x;
144             slice.rgt_z_handle.position.z = slice.z_particles.geometry.vertices[slice.rgt_z_handle_position].x;
145             /**
146              * slice bar
147              */
148             slice.create_slice_bar('z');
149         };
150     };
151 })();
