  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.SurfaceWorld requires JSurface3D');
  7     var last_vertex;
  8     /**
  9      * Implements any valid hover interaction with the surface.
 10      * Adds vertex_sphere, lines (tubes), active axis labels
 11      * @private
 12      * @param {Object} js3d An instance of JSurface3D
 13      * @param {THREE.Vector3} vertex The vertex within settings.snap_distance of the mouse
 14      * @param {Number} index The index of the vertex
 15      * @param {THREE.Mesh} object The closest object to the camera that THREE.Ray returned
 16      */
 17     var hover = function (js3d, vertex, index, object) {
 18         var hover_buffer = js3d.buffers.hover, hover_group = js3d.groups.hover, matlib = js3d.matlib,
 19             vertex_sphere = js3d.vertex_sphere, settings = js3d.settings;
 20         if (vertex === last_vertex) return;
 21         if (hover_group) js3d.groups.surface_top.remove(hover_group), hover_buffer.clear();
 22         last_vertex = vertex;
 23         hover_group = hover_buffer.add(new THREE.Object3D());
 24         hover_group.name = 'Hover Group';
 25         vertex_sphere.position.copy(vertex);
 26         vertex_sphere.position.y += js3d.settings.floating_height;
 27         vertex_sphere.updateMatrix();
 28         vertex_sphere.visible = true;
 29         (function () {
 30             // [xz]from & [xz]to are the start and end vertex indexes for a vertex row or column
 31             var x, xvertices = [], xvertices_bottom = [], z, zvertices = [], zvertices_bottom = [],
 32                 color = settings.interactive_surface_color,
 33                 xfrom = index - (index % (js3d.x_segments + 1)),
 34                 xto   = xfrom + js3d.x_segments + 1,
 35                 zfrom = index % (js3d.x_segments + 1),
 36                 zto   = ((js3d.x_segments + 1) * (js3d.z_segments + 1)) - (js3d.x_segments - zfrom);
 37             for (x = xfrom; x < xto; x++) xvertices.push(object.geometry.vertices[x]);
 38             for (z = zfrom; z < zto; z += js3d.x_segments + 1) zvertices.push(object.geometry.vertices[z]);
 39             js3d.hud.set_volatility(js3d.data.vol[index]);
 40             // surface lines
 41             (function () {
 42                 var xlft = js3d.x_segments - js3d.slice.lft_x_handle_position,
 43                     xrgt = js3d.x_segments - js3d.slice.rgt_x_handle_position,
 44                     zlft = js3d.z_segments - js3d.slice.lft_z_handle_position,
 45                     zrgt = js3d.z_segments - js3d.slice.rgt_z_handle_position,
 46                     zmin = Math.min.apply(null, [zlft, zrgt]),
 47                     zmax = Math.max.apply(null, [zlft, zrgt]),
 48                     xmin = Math.min.apply(null, [xlft, xrgt]),
 49                     xmax = Math.max.apply(null, [xlft, xrgt]),
 50                     x_surface_lines = new Four.Tube(matlib, xvertices.slice(xmin, xmax + 1), color),
 51                     z_surface_lines = new Four.Tube(matlib, zvertices.slice(zmin, zmax + 1), color);
 52                 x_surface_lines.name = 'X Surface Lines';
 53                 z_surface_lines.name = 'Z Surface Lines';
 54                 hover_group.add(x_surface_lines);
 55                 hover_group.add(z_surface_lines);
 56             }());
 57             // smile z, z and y lines
 58             (function () {
 59                 var yvertices = [{x: 0, y: 0, z: vertex.z}, {x: 0, y: settings.surface_y, z: vertex.z}],
 60                     zlines = new Four.Tube(matlib, zvertices, color), ylines = new Four.Tube(matlib, yvertices, color);
 61                 zlines.name = 'Smile Z, Z Hover Lines';
 62                 ylines.name = 'Smile Z, Y Hover Lines';
 63                 zlines.position.x = Math.abs(zvertices[0].x - settings.surface_x / 2) + settings.smile_distance;
 64                 ylines.position.x = settings.surface_x / 2 + settings.smile_distance;
 65                 zlines.matrixAutoUpdate = false;
 66                 ylines.matrixAutoUpdate = false;
 67                 zlines.updateMatrix();
 68                 ylines.updateMatrix();
 69                 hover_group.add(zlines);
 70                 hover_group.add(ylines);
 71             }());
 72             // smile x, x and y lines
 73             (function () {
 74                 var yvertices = [{x: vertex.x, y: 0, z: 0}, {x: vertex.x, y: settings.surface_y, z: 0}],
 75                     xlines = new Four.Tube(matlib, xvertices, color), ylines = new Four.Tube(matlib, yvertices, color);
 76                 xlines.name = 'Smile X, X Hover Lines';
 77                 ylines.name = 'Smile X, Y Hover Lines';
 78                 xlines.position.z = -((settings.surface_z / 2) + xvertices[0].z + settings.smile_distance);
 79                 ylines.position.z = -(settings.surface_z / 2) - settings.smile_distance;
 80                 xlines.matrixAutoUpdate = false;
 81                 ylines.matrixAutoUpdate = false;
 82                 xlines.updateMatrix();
 83                 ylines.updateMatrix();
 84                 hover_group.add(xlines);
 85                 hover_group.add(ylines);
 86             }());
 87             // bottom grid, x and z lines
 88             (function () {
 89                 var xlines, zlines;
 90                 xvertices_bottom.push(xvertices[0].clone(), xvertices[xvertices.length-1].clone());
 91                 xvertices_bottom[0].y = xvertices_bottom[1].y = -settings.floating_height;
 92                 zvertices_bottom.push(zvertices[0].clone(), zvertices[zvertices.length-1].clone());
 93                 zvertices_bottom[0].y = zvertices_bottom[1].y = -settings.floating_height;
 94                 xlines = new Four.Tube(matlib, xvertices_bottom, color);
 95                 zlines = new Four.Tube(matlib, zvertices_bottom, color);
 96                 xlines.name = 'Bottom Grid X Hover lines';
 97                 zlines.name = 'Bottom Grid Z Hover lines';
 98                 xlines.matrixAutoUpdate = false;
 99                 zlines.matrixAutoUpdate = false;
100                 hover_group.add(xlines);
101                 hover_group.add(zlines);
102             }());
103             // surface labels
104             ['x', 'z'].forEach(function (val) {
105                 var data = js3d.data, lbl_arr = data[val + 's_labels'] || data[val + 's'], width, offset, lbl, vertices,
106                     txt = val === 'x'
107                         ? lbl_arr[index % (js3d.x_segments + 1)]
108                         : lbl_arr[~~(index / (js3d.x_segments + 1))],
109                     scale = '0.1', group = new THREE.Object3D();
110                 group.name = 'Surface Label ' + val;
111                 // create label
112                 lbl = js3d.text3d(txt, color);
113                 lbl.matrixAutoUpdate = false;
114                 width = lbl.geometry.boundingSphere.radius / 2;
115                 offset = ((width / 2) * scale) + (width * 0.05); // half the width * scale + a relative offset
116                 vertices = val === 'x' ? zvertices : xvertices;
117                 group.add(lbl);
118                 // create box
119                 (function () {
120                     var height = 60,
121                         box_width = (width * 2) + 20,
122                         box = new THREE.CubeGeometry(box_width, height, 4, 4, 0, 0),
123                         mesh = new THREE.Mesh(box, matlib.get_material('phong', 0xdddddd));
124                     mesh.position.x = box_width / 2 - 10;
125                     mesh.position.y = 20;
126                     // create the tail by moving the 2 center vertices closes to the surface
127                     mesh.geometry.vertices.filter(function (val) {
128                         return (val.x === 0 && val.y === height / 2);
129                     }).forEach(function (vertex) {vertex.y = height;});
130                     mesh.matrixAutoUpdate = false;
131                     mesh.updateMatrix();
132                     group.add(mesh);
133                 }());
134                 // position / rotation
135                 group.scale.set(scale, scale, scale);
136                 group.position.y = -settings.floating_height + 0.5;
137                 group.rotation.x = -Math.PI * 0.5;
138                 if (val === 'x') {
139                     group.position.x = vertices[0][val] - offset;
140                     group.position.z = (settings.surface_z / 2) + 12 + settings.axis_offset;
141                     group.position.z = (settings.surface_z / 2) + 12 + settings.axis_offset;
142                 }
143                 if (val === 'z') {
144                     group.position.x = -((settings.surface_x / 2) + 12) - settings.axis_offset;
145                     group.position.z = vertices[0][val] - offset;
146                     group.rotation.z = -Math.PI * 0.5;
147                 }
148                 group.matrixAutoUpdate = false;
149                 group.updateMatrix();
150                 hover_group.matrixAutoUpdate = false;
151                 hover_group.updateMatrix();
152                 hover_group.add(group);
153             });
154         }());
155         js3d.groups.surface_top.add(js3d.groups.hover = hover_group);
156     };
157     /**
158      * Test if the cursor is over a mesh
159      * @event {Object} mouse event object
160      * @meshes {Array} meshes array of meshes to test
161      * @return {Object} THREE.Ray intersects object
162      */
163     var intersects_mesh = function (js3d, event, meshes) {
164         var mouse = {x: 0, y: 0}, vector, ray;
165         mouse.x = ((event.pageX - js3d.sel_offset.left) / js3d.width) * 2 - 1;
166         mouse.y = -((event.pageY - js3d.sel_offset.top) / js3d.height) * 2 + 1;
167         vector = new THREE.Vector3(mouse.x, mouse.y, 0.5);
168         js3d.projector.unprojectVector(vector, js3d.camera);
169         ray = new THREE.Ray(js3d.camera.position, vector.subSelf(js3d.camera.position).normalize());
170         return ray.intersectObjects(meshes);
171     };
172     /**
173      * Creates all geomerty, initializes data and implements interactions
174      * @name JSurface3D.SurfaceWorld
175      * @namespace JSurface3D.SurfaceWorld
176      * @param {Object} js3d JSurface3D instance
177      * @private
178      * @constructor
179      */
180     window.JSurface3D.SurfaceWorld = function (js3d) {
181         var surface_world = this, settings = js3d.settings, matlib = js3d.matlib, webgl = js3d.webgl;
182         /**
183          * Creates all the geomerty and groups, removes existing items first
184          * @name JSurface3D.SurfaceWorld.prototype.create_world
185          * @function
186          * @private
187          * @return {THREE.Object3D}
188          */
189         surface_world.create_world = function () {
190             var groups = js3d.groups, buffers = js3d.buffers;
191             if (groups.surface) groups.animation.remove(groups.surface), buffers.surface.clear();
192             js3d.interactive_meshes.remove('surface');
193             groups.surface = new THREE.Object3D();
194             groups.surface.name = 'Surface Group';
195             groups.slice = buffers.slice.add(new THREE.Object3D());
196             groups.slice.name = 'Slice Group';
197             groups.surface_top = new THREE.Object3D();
198             groups.surface_bottom = new THREE.Object3D();
199             if (typeof js3d.surface_plane === 'object') js3d.surface_plane.update();
200             else js3d.surface_plane.init();
201             groups.slice.add(js3d.surface_plane);
202             if (webgl) {
203                 // create bottom grid
204                 (function () {
205                     var grid = Four.multimaterial_object(
206                         JSurface3D.plane(js3d, 'surface'),
207                         matlib.get_material('compound_grid_wire')
208                     );
209                     grid.overdraw = true;
210                     grid.matrixAutoUpdate = false;
211                     groups.surface_bottom.add(buffers.surface.add(grid));
212                 })();
213                 // create floor
214                 (function () {
215                     var floor = Four.multimaterial_object(
216                         new THREE.PlaneGeometry(5000, 5000, 100, 100),
217                         matlib.get_material('compound_floor_wire')
218                     );
219                     floor.rotation.x = - Math.PI / 2;
220                     floor.position.y = -0.01;
221                     groups.surface_bottom.add(buffers.surface.add(floor));
222                 })();
223                 // create axes
224                 (function () {
225                     var x_axis = JSurface3D.Axis({axis: 'x'}, js3d),
226                         z_axis = JSurface3D.Axis({axis: 'z'}, js3d);
227                     x_axis.position.z = settings.surface_z / 2 + settings.axis_offset;
228                     z_axis.position.x = -settings.surface_x / 2 - settings.axis_offset;
229                     z_axis.rotation.y = -Math.PI * 0.5;
230                     groups.surface_bottom.add(x_axis);
231                     groups.surface_bottom.add(z_axis);
232                 })();
233                 groups.surface_top.add(JSurface3D.smile(js3d));
234                 groups.animation.add(js3d.vertex_sphere);
235                 groups.surface_top.position.y = js3d.settings.floating_height;
236                 groups.surface.add(groups.surface_bottom);
237             }
238             groups.surface.add(buffers.surface.add(groups.surface_top));
239             groups.surface.add(groups.slice);
240             return groups.surface;
241         };
242         /**
243          * Scale data to fit surface dimensions, apply Log (to x and z) if enabled
244          * @name JSurface3D.SurfaceWorld.prototype.init_data
245          * @param {Object} data
246          * @function
247          * @private
248          */
249         surface_world.init_data = function (data) {
250             var log = function (arr) {return arr.map(function (val) {return Math.log(val);});};
251             // adjusted data is the original data scaled to fit 2D grids width/length/height.
252             // It is used to set the distance between plane segments. Then the real data is used as the values
253             js3d.adjusted_vol = Four.scale(data.vol, 0, settings.surface_y);
254             js3d.adjusted_xs = Four.scale(
255                 js3d.settings.log ? log(data.xs) : data.xs,
256                 -(settings.surface_x / 2), settings.surface_x / 2
257             );
258             js3d.adjusted_zs = Four.scale(
259                 js3d.settings.log ? log(data.zs) : data.zs,
260                 -(settings.surface_z / 2), settings.surface_z / 2
261             );
262             // data.ys doesnt exist, so we need to create js3d.adjusted_ys manualy out of the given
263             // vol plane range: 0 - settings.surface_y
264             js3d.adjusted_ys = (function () {
265                 var increment = settings.surface_y / js3d.y_segments, arr = [], i;
266                 for (i = 0; i < js3d.y_segments + 1; i++) arr.push(i * increment);
267                 return arr;
268             }());
269             // create data.ys out of js3d.vol_min and js3d.vol_max, call it ys not to confuse with config paramater
270             // we are not using ys to create js3d.adjusted_ys as we want more control over js3d.adjusted_ys
271             // than a standard Four.scale
272             js3d.ys = (function () {
273                 var increment = (js3d.vol_max - js3d.vol_min) / js3d.y_segments, arr = [], i;
274                 for (i = 0; i < js3d.y_segments + 1; i++)
275                     arr.push((+js3d.vol_min + (i * increment)).toFixed(settings.precision_lbl));
276                 return arr;
277             }());
278         };
279         /**
280          * Implement interactivity
281          * @name JSurface3D.SurfaceWorld.prototype.interactive
282          * @function
283          * @private
284          */
285         surface_world.interactive = function () {
286             var imeshes = js3d.interactive_meshes.meshes, groups = js3d.groups,
287                 $selector = js3d.selector, surface_hittest, handle_hittest, // store the return of successful raycasts
288                 mousedown = false, sx = 0, sy = 0, mouse_x = null, mouse_y = null,
289                 hit_handle = false, rotation_enabled = true, slice_enabled = false;
290             /**
291              * Populate surface_hittest and handle_hittest
292              * Trigger rotate_world and slice_handle_drag events
293              */
294             $selector.on('mousemove.surface.interactive', function (event) {
295                 event.preventDefault();
296                 js3d.local_settings.play = true;
297                 js3d.local_settings.stopping = false;
298                 var xlft = intersects_mesh(js3d, event, [imeshes.lft_x_handle]),
299                     xrgt = intersects_mesh(js3d, event, [imeshes.rgt_x_handle]),
300                     zlft = intersects_mesh(js3d, event, [imeshes.lft_z_handle]),
301                     zrgt = intersects_mesh(js3d, event, [imeshes.rgt_z_handle]);
302                 /**
303                  * slice handle interactions
304                  */
305                 hit_handle = false;
306                 if (xlft.length) handle_hittest = xlft, handle_hittest[0].lbl = 'lft_x_handle', hit_handle = true;
307                 if (xrgt.length) handle_hittest = xrgt, handle_hittest[0].lbl = 'rgt_x_handle', hit_handle = true;
308                 if (zlft.length) handle_hittest = zlft, handle_hittest[0].lbl = 'lft_z_handle', hit_handle = true;
309                 if (zrgt.length) handle_hittest = zrgt, handle_hittest[0].lbl = 'rgt_z_handle', hit_handle = true;
310                 hit_handle ? $selector.trigger('handle_over') : $selector.trigger('handle_out');
311                 /**
312                  * surface interaction
313                  */
314                 surface_hittest = intersects_mesh(js3d, event, [imeshes.surface]);
315                 if (surface_hittest.length > 0 && surface_hittest[0].face.materialIndex === 1)
316                     $selector.trigger('surface_over', surface_hittest);
317                 else $selector.trigger('surface_out');
318                 /**
319                  * original mouse x & y
320                  */
321                 mouse_x = event.pageX;
322                 mouse_y = event.pageY;
323                 /**
324                  * Trigger custom events
325                  */
326                 if (mousedown && rotation_enabled) $selector.trigger('rotate_world', event);
327                 if (webgl && mousedown && slice_enabled) $selector.trigger('slice_handle_drag', event);
328             });
329             $selector.on('rotate_world', function () {
330                 var dx = mouse_x - sx, dy = mouse_y - sy;
331                 groups.animation.rotation.y += dx * 0.01;
332                 groups.animation.rotation.x += dy * 0.01;
333                 sx += dx, sy += dy;
334             });
335             $selector.on('mousedown.surface.interactive', function (event) {
336                 event.preventDefault();
337                 mousedown = true, sx = event.pageX, sy = event.pageY;
338                 if (hit_handle) $selector.trigger('slice_handle_click');
339                 $(document).on('mouseup.surface.interactive', function () {
340                     rotation_enabled = true;
341                     slice_enabled = false;
342                     mousedown = false;
343                     $(document).off('mouse.surface.interactive');
344                 });
345             });
346             $selector.on('mousewheel.surface.interactive', function (event, direction) {
347                 if (!direction) return; // jquery mousewheel plugin doesnt exist
348                 var camera = js3d.camera, pos = camera.position.z - (direction * settings.zoom_sensitivity);
349                 camera.position.z = pos < 1 ? 1 : pos;
350                 camera.lookAt({x: 0, y: 0, z: 0});
351             });
352             /**
353              * Only implement hover interactivity for webgl browsers
354              */
355             if (!webgl) return;
356             $selector.on('surface_over', function (event, intersects) {
357                 var faces = 'abcd', i = 0, index, vertex, vertex_world_position,
358                     intersected_obj = $.isArray(intersects) ? intersects[0] : intersects,
359                     object = intersected_obj.object, point = intersected_obj.point;
360                 for (; i < 4; i++) { // loop through vertices
361                     index = intersected_obj.face[faces.charAt(i)];
362                     vertex = object.geometry.vertices[index];
363                     vertex_world_position = object.matrixWorld.multiplyVector3(vertex.clone());
364                     if (vertex_world_position.distanceTo(point) < settings.snap_distance) {
365                         hover(js3d, vertex, index, object);
366                     }
367                 }
368             });
369             $selector.on('surface_out', function () {
370                 if (groups.hover) {
371                     groups.surface_top.remove(groups.hover);
372                     last_vertex = null;
373                     js3d.buffers.hover.clear();
374                 }
375                 js3d.vertex_sphere.visible = false;
376                 js3d.hud.set_volatility();
377             });
378             $selector.on('handle_over', function () {
379                 js3d.slice.reset_handle_material();
380                 handle_hittest[0].object.material = matlib.get_material('phong', settings.slice_handle_color_hover);
381                 $selector.css({cursor: 'pointer'});
382             });
383             $selector.on('handle_out', function () {
384                 js3d.slice.reset_handle_material();
385                 $selector.css({cursor: 'default'});
386             });
387             $selector.on('slice_handle_click', function () {
388                 slice_enabled = true;
389                 rotation_enabled = false;
390             });
391             /**
392              * Move the drag handles and update the bar
393              */
394             $selector.on('slice_handle_drag', function (event, original_event) {
395                 var handle_lbl = handle_hittest[0].lbl, axis = handle_lbl.replace(/^.*_([xz])_.*$/, '$1'),
396                     particles = js3d.slice[axis + '_particles'];
397                 (function () {
398                     var intersects = intersects_mesh(js3d, original_event, [js3d.slice.intersection_plane]),
399                         vertices = particles.geometry.vertices, vertex, vertex_world_position, index,
400                         i = vertices.length, dist = [] /* distances from raycast/plane intersection & particles */;
401                     while (i--) {
402                         vertex = vertices[i];
403                         vertex_world_position = particles.matrixWorld.multiplyVector3(vertices[i].clone());
404                         dist[i] = vertex_world_position.distanceTo(intersects[0].point);
405                     }
406                     index = dist.indexOf(Math.min.apply(null, dist));
407                     if (index !== js3d.slice[handle_lbl + '_position']) {
408                         js3d.slice.reset_handle_material();
409                         handle_hittest[0].object.material = matlib
410                             .get_material('phong', settings.slice_handle_color_hover);
411                         /**
412                          * move handles along particles
413                          */
414                         js3d.slice[handle_lbl + '_position'] = index;
415                         js3d.slice[handle_lbl].position[axis] = vertices[index].x;
416                         /**
417                          * recreate surface plane
418                          */
419                         js3d.surface_plane.update();
420                         /**
421                          * update resize bars
422                          */
423                         js3d.slice.create_slice_bar(axis);
424                     }
425                 }());
426             });
427         };
428     };
