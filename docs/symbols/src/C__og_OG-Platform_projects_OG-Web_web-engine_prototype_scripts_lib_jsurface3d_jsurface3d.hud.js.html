  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.Hud requires JSurface3D');
  7     /**
  8      * Create the html overlay with the log option and volatility display
  9      * @param {Object} js3d A JSurface3D instance
 10      * @param {String} $selector A Jquery selector to the surfaces container
 11      * @param {String} stylesheet Selector to the surfaces shared stylesheet location
 12      * @name JSurface3D.Hud
 13      * @namespace JSurface3D.Hud
 14      * @private
 15      * @constructor
 16      */
 17     window.JSurface3D.Hud = function (js3d, $selector, stylesheet) {
 18         var hud = this, settings = js3d.settings;
 19         /**
 20          * Loads 2D overlay display with form
 21          * @name JSurface3D.Hud.load
 22          * @function
 23          * @private
 24          */
 25         hud.load = function () {
 26             if (!settings.hud) return;
 27             $selector.find('.js3d').remove();
 28             (function () {
 29                 if ($(stylesheet).length) return;
 30                 var css = '\
 31                     .js3d {bottom: 10px; left: 10px; top: 0; position: absolute;}\
 32                     .js3d .j-o {position: absolute; top: 10px; white-space: nowrap;}\
 33                     .js3d .j-o input {vertical-align: top;}\
 34                     .js3d .j-v {padding-top: 9px; padding-bottom: 3px; position: absolute; bottom: 0;}\
 35                     .js3d .j-v canvas {border: 1px solid #ccc;}\
 36                     .js3d .j-v span {position: absolute; left: 20px;}\
 37                     .js3d .j-v .j-max {top: 0;}\
 38                     .js3d .j-v .j-min {bottom: 0;}\
 39                     .js3d .j-v .j-vol {display: none; background: #eee; border: 1px solid #ddd; padding: 0 5px;}\
 40                     .js3d .j-v .j-vol:after {content: ""; display: block; position: absolute; \
 41                         width: 0; height: 0; left: -10px; top: 50%; margin-top: -4px; \
 42                         border-top: 5px solid transparent; border-right: 10px solid #ddd; \
 43                         border-bottom: 5px solid transparent;}',
 44                     head = document.querySelector('head'), style = document.createElement('style');
 45                 style.setAttribute('data-og', 'surface');
 46                 if (style.styleSheet) style.styleSheet.cssText = css; // IE
 47                 else style.appendChild(document.createTextNode(css));
 48                 head.appendChild(stylesheet = style);
 49             })();
 50             var tmpl = '\
 51                 <div class="js3d">\
 52                   <div class="j-o"><label>Log<input type="checkbox" checked="checked" /></label></div>\
 53                   <div class="j-v">\
 54                     <span class="j-max">{{max}}</span><span class="j-min">{{min}}</span><span class="j-vol"></span>\
 55                     <canvas>canvas</canvas>\
 56                   </div>\
 57                 </div>',
 58                 min = js3d.vol_min.toFixed(settings.precision_hud), max = js3d.vol_max.toFixed(settings.precision_hud),
 59                 html = tmpl.replace(/\{\{(?:max|min)\}\}/g, function (m) {return m === '{{min}}' ? min : max;}),
 60                 $html = $(html).appendTo($selector);
 61             hud.vol_canvas_height = js3d.height / 2;
 62             if (js3d.webgl) hud.volatility($html.find('canvas')[0]);
 63             else $html.find('.j-v').hide();
 64             hud.form();
 65         };
 66         /**
 67          * Attach change handler to log option
 68          * @function
 69          * @private
 70          * @ignore
 71          */
 72         hud.form = function () {
 73             $selector.find('.j-o input').prop('checked', js3d.settings.log).on('change', function () {
 74                 js3d.settings.log = $(this).is(':checked');
 75                 js3d.update('world');
 76             });
 77         };
 78         /**
 79          * Set a value in the 2D volatility display
 80          * @param {Number} value The value to set. If value is not a number the indicator is hidden
 81          * @function
 82          * @private
 83          * @ignore
 84          */
 85         hud.set_volatility = function (value) {
 86             if (!settings.hud) return;
 87             var top = (Four.scale([js3d.vol_min, value, js3d.vol_max], hud.vol_canvas_height, 0))[1],
 88                 css = {top: top + 'px', color: settings.interactive_hud_color},
 89                 $vol = $selector.find('.j-vol');
 90             typeof value === 'number'
 91                 ? $vol.html(value.toFixed(settings.precision_hud)).css(css).show()
 92                 : $vol.empty().hide();
 93         };
 94         hud.vol_canvas_height = null;
 95         /**
 96          * Volatility gradient canvas
 97          * @param {Object} canvas html canvas element
 98          * @function
 99          * @private
100          * @ignore
101          */
102         hud.volatility = function (canvas) {
103             var ctx = canvas.getContext('2d'), gradient,
104                 min_hue = settings.vertex_shading_hue_min, max_hue = settings.vertex_shading_hue_max,
105                 steps = Math.abs(max_hue - min_hue) / 60, stop;
106             gradient = ctx.createLinearGradient(0, 0, 0, hud.vol_canvas_height);
107             canvas.width = 10;
108             canvas.height = hud.vol_canvas_height;
109             gradient.addColorStop(0, 'hsl(' + max_hue + ', 100%, 50%)');
110             while (steps--) { // 60: the number of hue increaments before one color starts fading in and another out
111                 stop = (steps * 60 / (min_hue / 100)) / 100;
112                 gradient.addColorStop(stop, 'hsl(' + steps * 60 + ', 100%, 50%)');
113                 gradient.addColorStop(stop, 'hsl(' + steps * 60 + ', 100%, 50%)');
114             }
115             gradient.addColorStop(1, 'hsl(' + min_hue + ', 100%, 50%)');
116             ctx.fillStyle = gradient;
117             ctx.fillRect(0, 0, 10, hud.vol_canvas_height);
118         };
119     };
120 })();
