  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.Smile requires JSurface3D');
  7     /**
  8      * Creates new smile planes
  9      * @name JSurface3D.smile
 10      * @namespace JSurface3D.smile
 11      * @param {Object} js3d JSurface3D instance
 12      * @function
 13      * @private
 14      * @returns {THREE.Object3D}
 15      */
 16     window.JSurface3D.smile = function (js3d) {
 17         var settings = js3d.settings, matlib = js3d.matlib, smile = new THREE.Object3D();
 18         var x = function () {
 19             var obj = new THREE.Object3D();
 20             (function () { // plane
 21                 var plane = JSurface3D.plane(js3d, 'smilex'),
 22                     material = matlib.get_material('compound_grid_wire'),
 23                     mesh = Four.multimaterial_object(plane, material);
 24                 mesh.rotation.x = Math.PI * 0.5;
 25                 mesh.position.y = settings.surface_y;
 26                 mesh.position.z = -((settings.surface_z / 2) + settings.smile_distance);
 27                 mesh.matrixAutoUpdate = false;
 28                 mesh.updateMatrix();
 29                 obj.add(mesh);
 30             }());
 31             (function () { // axis
 32                 var y_axis = JSurface3D.Axis({axis: 'y', right: true}, js3d);
 33                 y_axis.position.x = -(settings.surface_x / 2) - 25;
 34                 y_axis.position.y = 4;
 35                 y_axis.position.z = -(settings.surface_z / 2) - settings.smile_distance;
 36                 y_axis.rotation.y = Math.PI * 0.5;
 37                 y_axis.rotation.z = Math.PI * 0.5;
 38                 obj.add(y_axis);
 39             }());
 40             return obj;
 41         };
 42         var z = function () {
 43             var obj = new THREE.Object3D();
 44             (function () { // plane
 45                 var plane = JSurface3D.plane(js3d, 'smiley'),
 46                     material = matlib.get_material('compound_grid_wire'),
 47                     mesh = Four.multimaterial_object(plane, material);
 48                 mesh.position.x = (settings.surface_x / 2) + settings.smile_distance;
 49                 mesh.rotation.z = Math.PI * 0.5;
 50                 mesh.matrixAutoUpdate = false;
 51                 mesh.updateMatrix();
 52                 obj.add(mesh);
 53             }());
 54             (function () { // axis
 55                 var y_axis = JSurface3D.Axis({axis: 'y'}, js3d);
 56                 y_axis.position.y = 4;
 57                 y_axis.position.z = (settings.surface_z / 2) + 5;
 58                 y_axis.position.x = (settings.surface_x / 2) + settings.smile_distance;
 59                 y_axis.rotation.z = Math.PI * 0.5;
 60                 obj.add(y_axis);
 61             }());
 62             return obj;
 63         };
 64         var shadows = function () {
 65             var obj = new THREE.Object3D();
 66             (function () { // x shadow
 67                 var z = settings.surface_z / 2 + settings.smile_distance, half_width = settings.surface_x / 2,
 68                     points = [{x: -half_width, y: 0, z: -z}, {x: half_width, y: 0, z: -z}],
 69                     shadow = new Four.Tube(matlib, points, 0xaaaaaa);
 70                 shadow.matrixAutoUpdate = false;
 71                 obj.add(shadow);
 72             }());
 73             (function () { // z shadow
 74                 var x = settings.surface_x / 2 + settings.smile_distance, half_width = settings.surface_z / 2,
 75                     points = [{x: x, y: 0, z: -half_width}, {x: x, y: 0, z: half_width}],
 76                     shadow = new Four.Tube(matlib, points, 0xaaaaaa);
 77                 shadow.matrixAutoUpdate = false;
 78                 obj.add(shadow);
 79             }());
 80             obj.position.y -= settings.floating_height;
 81             return obj;
 82         };
 83         smile.add(x());
 84         smile.add(z());
 85         smile.add(shadows());
 86         return smile;
 87     };
 88 })();
