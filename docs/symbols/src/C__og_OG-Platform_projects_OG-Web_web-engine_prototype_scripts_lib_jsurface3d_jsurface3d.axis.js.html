  5 (function () {
  6     if (!window.JSurface3D) throw new Error('JSurface3D.Axis requires JSurface3D');
  7     /**
  8      * Remove every nth item in Array keeping the first and last,
  9      * also specifically remove the second last (as we want to keep the last)
 10      * @private
 11      * @param {Array} arr
 12      * @param {Number} nth
 13      * @returns {Array}
 14      */
 15     var thin = function (arr, nth) {
 16         if (!nth || nth === 1) return arr;
 17         var len = arr.length;
 18         return arr.filter(function (val, i) {
 19             return ((i === 0) || !(i % nth) || (i === (len -1))) || (i === (len -2)) && false
 20         });
 21     };
 22     /**
 23      * Creates an Axis with labels for the bottom grid
 24      * @name JSurface3D.Axis
 25      * @namespace JSurface3D.Axis
 26      * @param {Object} config
 27      * <pre>
 28      *     axis     {String} x or z
 29      *     spacing  {Array}  Array of numbers adjusted to fit units of mesh
 30      *     labels   {Array}  Array of labels
 31      *     label    {String} Axis label
 32      * </pre>
 33      * @param {Object} js3d A JSurface3D instance
 34      * @private
 35      * @return {THREE.Object3D}
 36      */
 37     window.JSurface3D.Axis = function (config, js3d) {
 38         var axis = config.axis, spacing = js3d['adjusted_' + axis + 's'],
 39             values = js3d.data[config.axis + 's_labels'] || (axis === 'y' ? js3d[axis + 's'] : js3d.data[axis + 's']),
 40             lbl = axis === 'y' ? null : js3d.data[axis + 's_label'],
 41             mesh = new THREE.Object3D(), i, nth = Math.ceil(spacing.length / 6),
 42             lbl_arr = thin(values, nth), pos_arr = thin(spacing, nth), settings = js3d.settings,
 43             axis_len = settings['surface_' + axis];
 44         mesh.name = axis + ' Axis';
 45         (function () { // axis values
 46             var value, n;
 47             for (i = 0; i < lbl_arr.length; i++) {
 48                 n = lbl_arr[i];
 49                 n = n % 1 === 0 || isNaN(n % 1) ? n : (+n).toFixed(settings.precision_lbl).replace(/0+$/, '');
 50                 value = js3d.text3d(n+'', settings.font_color);
 51                 value.scale.set(0.1, 0.1, 0.1);
 52                 if (config.axis === 'y') {
 53                     value.position.x = pos_arr[i] - 6;
 54                     value.position.z = config.right
 55                         ? -(THREE.FontUtils.drawText(n).offset * 0.2) + 18
 56                         : 2;
 57                     value.rotation.z = -Math.PI * 0.5;
 58                     value.rotation.x = -Math.PI * 0.5;
 59                 } else {
 60                     value.rotation.x = -Math.PI * 0.5;
 61                     value.position.x = pos_arr[i] - ((THREE.FontUtils.drawText(n).offset * 0.2) / 2);
 62                     value.position.y = 0.1;
 63                     value.position.z = 12;
 64                 }
 65                 value.matrixAutoUpdate = false;
 66                 value.updateMatrix();
 67                 js3d.buffers.load.add(value);
 68                 mesh.add(value);
 69             }
 70         }());
 71         (function () { // axis label
 72             if (!lbl) return;
 73             var label = js3d.text3d(lbl, settings.font_color_axis_labels, true, true);
 74             label.scale.set(0.2, 0.2, 0.1);
 75             label.rotation.x = -Math.PI * 0.5;
 76             label.position.x = -(axis_len / 2) -3;
 77             label.position.y = 1;
 78             label.position.z = 22;
 79             label.matrixAutoUpdate = false;
 80             label.updateMatrix();
 81             js3d.buffers.load.add(label);
 82             mesh.add(label);
 83         }());
 84         (function () { // axis ticks
 85             var canvas = document.createElement('canvas'),
 86                 ctx = canvas.getContext('2d'),
 87                 plane = new THREE.PlaneGeometry(axis_len, 5, 0, 0),
 88                 axis = new THREE.Mesh(plane, js3d.matlib.texture(new THREE.Texture(canvas))),
 89                 tick_stop_pos =  settings.tick_length + 0.5,
 90                 labels = thin(spacing.map(function (val) {
 91                     // if not y axis offset half. y planes start at 0, x and z start at minus half width
 92                     var offset = config.axis === 'y' ? 0 : axis_len / 2;
 93                     return (val + offset) * (settings.texture_size / axis_len);
 94                 }), nth);
 95             plane.applyMatrix(new THREE.Matrix4().makeRotationX(-Math.PI / 2));
 96             canvas.width = settings.texture_size;
 97             canvas.height = 32;
 98             ctx.beginPath();
 99             ctx.lineWidth = 2;
100             for (i = 0; i < labels.length; i++)
101                 ctx.moveTo(labels[i] + 0.5, tick_stop_pos), ctx.lineTo(labels[i] + 0.5, 0);
102             ctx.moveTo(0.5, tick_stop_pos);
103             ctx.lineTo(0.5, 0.5);
104             ctx.lineTo(canvas.width - 0.5, 0.5);
105             ctx.lineTo(canvas.width - 0.5, tick_stop_pos);
106             ctx.stroke();
107             axis.material.map.needsUpdate = true;
108             axis.doubleSided = true;
109             if (config.axis === 'y') {
110                 if (config.right) axis.rotation.x = Math.PI, axis.position.z = 20;
111                 axis.position.x = (axis_len / 2) - 4;
112             } else {
113                 axis.position.z = 5;
114             }
115             axis.matrixAutoUpdate = false;
116             axis.updateMatrix();
117             js3d.buffers.surface.add(axis);
118             mesh.add(axis);
119         }());
120         return mesh;
121     };
122 })();
